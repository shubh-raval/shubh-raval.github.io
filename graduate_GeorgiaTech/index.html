Manipulability Optimization for Inverse Kinematics Of 3R
import numpy as np
              from scipy.optimize import minimize
              import matplotlib.pyplot as plt
              
              def manipulability(theta1, theta2, theta3):
              
                  if np.isnan(theta1) or np.isnan(theta2) or np.isnan(theta3):
                      manipulability_index = 0
                      return manipulability_index
                  a1, a2, a3 = 0.2, 0.15, 0.12
                  
                  J11 = -a1*np.sin(theta1) - a2*np.sin(theta1+theta2) - a3*np.sin(theta1+theta2+theta3)
                  J12 = -a2*np.sin(theta1+theta2) - a3*np.sin(theta1+theta2+theta3)
                  J13 = -a3*np.sin(theta1+theta2+theta3)
                  J21 = a1*np.cos(theta1) + a2*np.cos(theta1+theta2) + a3*np.cos(theta1+theta2+theta3)
                  J22 = a2*np.cos(theta1+theta2) + a3*np.cos(theta1+theta2+theta3)
                  J23 = a3*np.cos(theta1+theta2+theta3)
                  
                  J = np.array([[J11, J12, J13], [J21, J22, J23], [1, 1, 1]])
                  
                  JJT = J @ J.T  
                  manipulability_index = np.sqrt(np.linalg.det(JJT))
                  
                  return manipulability_index
              
              def inversekinematics(thetaorientation, x, y, elbowup):
                  a1, a2, a3 = 0.2, 0.15, 0.12
              
                  x3pos = x - a3*np.cos(thetaorientation)
                  y3pos = y - a3*np.sin(thetaorientation)
                  
                  p13 = np.sqrt(x3pos**2 + y3pos**2)
                  C2inv = (p13**2 - a1**2 - a2**2)/(2*a1*a2)
                  
                  C2 = C2inv
                  
                  theta2invp = np.arctan2(np.sqrt(1-((p13**2-a1**2-a2**2)/(2*a1*a2))**2), (p13**2-a1**2-a2**2)/(2*a1*a2))
                  theta2invm = -np.arctan2(np.sqrt(1-((p13**2-a1**2-a2**2)/(2*a1*a2))**2), (p13**2-a1**2-a2**2)/(2*a1*a2))
                  
                  if elbowup:
                      theta2inv = theta2invp  # choose one
                  else:
                      theta2inv = theta2invm
              
                  S2 = np.sin(theta2inv)
                  
                  A = np.array([[a1+a2*C2, -a2*S2], [a2*S2, a1+a2*C2]])
                  b = np.array([x3pos, y3pos])
                  tmp = np.linalg.inv(A).dot(b)
                  
                  theta1inv = np.arctan2(tmp[1], tmp[0])
                  
                  theta3inv = thetaorientation - theta1inv - theta2inv
                  manipulability_score = manipulability(theta1inv, theta2inv, theta3inv)
                  
                  return theta1inv, theta2inv, theta3inv, manipulability_score
              
              def gradientdescent(x, y):
                  best_cost = float('inf')
                  best_result = None
                  best_theta = None
              
                  for theta_init in np.linspace(0, 2*np.pi, 64, endpoint=False):
                      res = minimize(
                          lambda theta: -max(
                              inversekinematics(theta[0], x, y, elbowup=True)[3],
                              inversekinematics(theta[0], x, y, elbowup=False)[3]
                          ),
                          x0=[theta_init],
                          bounds=[(0, 2 * np.pi)],
                          method='L-BFGS-B'
                      )
                      
                      if res.fun < best_cost:
                          best_cost = res.fun
                          best_theta = res.x[0]
                          r1 = inversekinematics(best_theta, x, y, elbowup=True)
                          r2 = inversekinematics(best_theta, x, y, elbowup=False)
                          best_result = r1 if r1[3] > r2[3] else r2
                  
              
                  return best_result + (best_theta,)
